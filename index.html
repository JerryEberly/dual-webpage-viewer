        .container {

            display: flex;          /* Makes the frames sit side by side */

            height: 90vh;          /* Takes up most of the screen height */

        }

        iframe {

            width: 50%;            /* Each frame gets half the width */

            height: 100%;          /* Fills the container height */

            border: none;          /* No ugly borders */

        }

        .advertisement {

            height: 10vh;          /* Small space for the ad */

            text-align: center;    /* Centers the ad text */

            background-color: #f0f0f0; /* Light gray background */

            display: flex;         /* Helps center the ad vertically */

            align-items: center;   /* Vertical centering */

            justify-content: center; /* Horizontal centering */

        }

    </style>

        <iframe id="leftFrame"></iframe>

        <p>Advertisement: <a href="https://example.com">Visit Example.com</a></p>

    </div>

    <script>

        // Function to grab a cookie by its name

        function getCookie(name) {

            let cookieArr = document.cookie.split(";");

            for(let i = 0; i < cookieArr.length; i++) {

                let cookiePair = cookieArr[i].split("=");

                if(name == cookiePair[0].trim()) {

                    return decodeURIComponent(cookiePair[1]);

                }

            }

            return null;

        }


        // Check if we already have URLs saved in cookies

        let leftUrl = getCookie("leftUrl");

        let rightUrl = getCookie("rightUrl");


        // If no URLs are saved, ask the user for them

        if (!leftUrl || !rightUrl) {

            leftUrl = prompt("Enter the URL for the left webpage (e.g., https://google.com):");

            rightUrl = prompt("Enter the URL for the right webpage (e.g., https://wikipedia.org):");

            // Save the URLs in cookies for 1 year

            document.cookie = "leftUrl=" + encodeURIComponent(leftUrl) + "; max-age=" + (365*24*60*60) + "; path=/";

            document.cookie = "rightUrl=" + encodeURIComponent(rightUrl) + "; max-age=" + (365*24*60*60) + "; path=/";

        }


        // Load the URLs into the frames

        document.getElementById("leftFrame").src = leftUrl;

        document.getElementById("rightFrame").src = rightUrl;

    </script>

</body>
